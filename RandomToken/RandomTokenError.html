def initialize(error, info = {})
  @code = error
  @info = info
  if ERRORS.keys.include?(error)
    @value = ERRORS[error][:value]
    @msg = ERRORS[error][:msg]
  elsif error.class.name == 'String'
    @code = :internal
    @value = 90000
    @msg = error
  else
    @code = :internal
    @value = 99999
    @msg = "Internal Error"
  end
end
